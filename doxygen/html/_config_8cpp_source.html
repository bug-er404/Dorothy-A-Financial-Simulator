170 {
171  // Flag if comment is present in the line

175 
176  // Check for valid configuration line

178  || (std::count(a_line.begin(), a_line.end(), '=') > 1))
179  return false;
180 
181  // Check for comments

183  {
184  // Skip whitespace and check if the line is only a comment
185 // trimBlanks(a_line);
186  if(a_line[0] == '#')
187  return false;
188 
189  t_commentFlag = true;
190  }
191 
192  // Get first half, remove whitespace and make upper.
193  a_name = a_line.substr(0, t_idx);
194  trimBlanks(a_name);
195  transform(a_name.begin(), a_name.end(), a_name.begin(), ::toupper);
196 
197 
198  // Get second half and remove white space
199  a_value = a_line.substr(t_idx+1, (a_line.size()-t_idx));
200  trimBlanks(a_value);
201 
202  // Check for trailing comments and removing it
203  if(t_commentFlag)
204  {
205  t_idx = a_value.find("#");
206  a_value = a_value.substr(0, t_idx);
207  }
