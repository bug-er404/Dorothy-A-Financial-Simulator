159 void TickerData::calculateMACDfields()
160 {
161  // MACD parameters:
162  const int FAST_EMA_VAL = 12;
163  const int SLOW_EMA_VAL = 26;
164  const int SIG_EMA_VAL = 9;
165 
166  // Factor values:
167  double t_k12 = ((2.0/FAST_EMA_VAL)+1.0);
168  double t_k26 = ((2.0/SLOW_EMA_VAL)+1.0);
169  double t_k9 = ((2.0/SIG_EMA_VAL)+1.0);
170 
171  // Iterators and temp vars:
172  int i;
173 
174  // FAST EMA CALCULATION
175  double t_sma12=0;
176  for(i=0;i<FAST_EMA_VAL;i++)
177  t_sma12 += m_tickerData[TICKER_FIELDS::CLOSE].at(i);
178  t_sma12 /= FAST_EMA_VAL;
179  // Squaring and filling in data
180  for(i=0; i<m_tickerData[0].size(); i++)
181  {
182  // EMA
183  if (i >= FAST_EMA_VAL)
184  m_tickerData[FAST_EMA].push_back(
185  ((m_tickerData[TICKER_FIELDS::CLOSE].at(i) - m_tickerData[TICKER_FIELDS::FAST_EMA].at(i-1)) * t_k12) + m_tickerData[TICKER_FIELDS::FAST_EMA].at(i-1));
186  // First SMA
187  else if (i==(FAST_EMA_VAL-1))
188  m_tickerData[TICKER_FIELDS::FAST_EMA].push_back(t_sma12);
189  // Filler value
190  else
191  m_tickerData[TICKER_FIELDS::FAST_EMA].push_back(NA_VAL);
192  }
193 
194  // SLOW EMA AND MACD LINE CALCULATION
195  double t_sma26=0;
196  for(i=0;i<SLOW_EMA_VAL;i++)
197  t_sma26 += m_tickerData[TICKER_FIELDS::CLOSE].at(i);
198  t_sma26 /= SLOW_EMA_VAL;
199  // Squaring and filling in datas
200  for(i=0; i<m_tickerData[0].size(); i++)
201  {
202  if (i >= SLOW_EMA_VAL)
203  {
204  // SLOW EMA
205  m_tickerData[TICKER_FIELDS::SLOW_EMA].push_back(
206  ((m_tickerData[TICKER_FIELDS::CLOSE].at(i) - m_tickerData[TICKER_FIELDS::SLOW_EMA].at(i-1)) * t_k26) + m_tickerData[TICKER_FIELDS::SLOW_EMA].at(i-1));
207  // MACD LINE
208  m_tickerData[TICKER_FIELDS::MACD_LINE].push_back(m_tickerData[TICKER_FIELDS::FAST_EMA].at(i)-m_tickerData[SLOW_EMA].at(i));
209  }
210  else if (i==(SLOW_EMA_VAL-1))
211  {
212  // SMA 26 and MACD LINE
213  m_tickerData[TICKER_FIELDS::SLOW_EMA].push_back(t_sma26);
214  m_tickerData[TICKER_FIELDS::MACD_LINE].push_back(m_tickerData[TICKER_FIELDS::FAST_EMA].at(i)-m_tickerData[TICKER_FIELDS::SLOW_EMA].at(i));
215  }
216  else {
217  m_tickerData[TICKER_FIELDS::SLOW_EMA].push_back(NA_VAL);
218  m_tickerData[TICKER_FIELDS::MACD_LINE].push_back(NA_VAL);
219  }
220  }
221 
222  // SIGNAL AND HISTOGRAM CALCULATION:
223  double t_sma9=0;
224  for(i=25;i<34;i++)
225  t_sma9 += m_tickerData[TICKER_FIELDS::MACD_LINE].at(i);
226  t_sma9 /= SIG_EMA_VAL;
227  for(i=0; i<m_tickerData[0].size(); i++)
228  {
229  if (i >= 34)
230  {
231  // SIGNAL LINE
232  m_tickerData[TICKER_FIELDS::SIG_LINE].push_back(
233  ((m_tickerData[TICKER_FIELDS::MACD_LINE].at(i) - m_tickerData[TICKER_FIELDS::SIG_LINE].at(i-1)) * t_k9) +
234  m_tickerData[TICKER_FIELDS::SIG_LINE].at(i-1));
235  // HISTOGRAM
236  m_tickerData[TICKER_FIELDS::MACD_HIST].push_back(m_tickerData[TICKER_FIELDS::MACD_LINE].at(i)-m_tickerData[TICKER_FIELDS::SIG_LINE].at(i));
237  }
238  else if (i==33)
239  {
240  m_tickerData[TICKER_FIELDS::SIG_LINE].push_back(t_sma9);
241  m_tickerData[TICKER_FIELDS::MACD_HIST].push_back(m_tickerData[TICKER_FIELDS::MACD_LINE].at(i)-m_tickerData[TICKER_FIELDS::SIG_LINE].at(i));
242  }
243  else{
244  m_tickerData[TICKER_FIELDS::SIG_LINE].push_back(NA_VAL);
