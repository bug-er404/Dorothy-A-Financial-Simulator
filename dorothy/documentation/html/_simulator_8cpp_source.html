75 void Simulator::RunSimulation()
76 {
77  // Update simulation counter
78  m_simulationCount++;
79 
80  std::cout<<"Dorothy: Running simulation..."<<std::endl;
81 
82  std::vector<std::string> t_tickers = m_data->GetUniverse();

84  std::cout << "Dorothy: Simulating for " << t_tradingDates.size() <<" days."<<std::endl;
85 
86  m_totalPNL = 0;
87 
88  // Set trading date index for respective tickers to run simulation:
89  for (auto &t_stock : m_tradingStocks) {
90  TickerData &t_stockData = *(m_data->GetTickerData(t_stock->GetStockName()));
91  t_stockData.SetTradingDateRange(t_tradingDates.at(0), t_tradingDates.at(t_tradingDates.size()-1));
92  }
93 
94  // * DATE RANGE LOOP *
95  for (int t_day = 0; t_day < t_tradingDates.size(); t_day++)
96  {
97  // Reset daily data variables:
98  m_today = t_tradingDates[t_day];
99  m_totalPosCount = 0;
100  m_shortPosCount = 0;
101  m_longPosCount = 0;
102  m_dailyPNL = 0.0;
103  m_ROR = 0.0;
104  m_grossMarketValue = 0.0;
105  m_netMarketValue = 0.0;
106 
107  // * CONSTITUENTS LOOP *
108  for (auto &t_stock : m_tradingStocks)
109  {
110  TickerData &t_stockData = *(m_data->GetTickerData(t_stock->GetStockName()));
111 
112  // Get date offset in the database from first date in data to trading start date
113  // We get the exact index of the data row.
114  double t_signal = calculateSignal(t_stockData);
115 
116  // Record signal
117  t_stock->RecordSignal(t_signal);
118 
119  // Write signal info to signal publication file
120  if(m_signalPublication)
121  recordSignalInfo(t_signal);
122 
123  // No signal:
124  if (t_signal == TickerData::NA_VAL)
125  {
126  // Record no transaction
127  recordNoActivity(*t_stock);
128  continue;
129  }
130  else
131  {
132  // Get closing price and make trade
133  double t_currentPrice = t_stockData.GetCurrentDataOf(TICKER_FIELDS::CLOSE);
134  makeTrade(*t_stock, t_currentPrice, t_signal);
135 
136  // Total, short and long positions in stock
137  double t_positions = t_stock->GetTotalShares();
138  if (t_stock->IsShort())
139  m_shortPosCount += t_stock->GetShortSharesHeld();
140  if (t_stock->IsLong())
141  m_longPosCount += t_stock->GetLongSharesHeld();
142 
143  // Update simulation variables:
144  m_totalPosCount += (t_stock->GetLongSharesHeld()+(-t_stock->GetShortSharesHeld()));
145  m_dailyPNL += t_stock->GetDailyPnL();
146  m_totalPNL += t_stock->GetDailyPnL();
147  m_ROR += t_stock->GetDailyROR();
148  m_grossMarketValue += abs(t_positions*t_currentPrice);
149  m_netMarketValue += (t_stock->GetLongCapital()-t_stock->GetShortCapital());
150  }
151  }
152  // Record daily PNL and ROR:
153  m_dailyPNLRecord.push_back(m_dailyPNL);
154  m_dailyCumulativeROR.push_back(m_ROR);
155 
156  // Write info to daily statistics file if flag is set.
157  if(m_dailyPublication)
158  recordDailyStatistics();
159  }
160

163  std::cout<< "Dorothy run statistics: "<<std::endl;

166

171  if(m_dailyPublication)

175  // Write the monthly statistics file and close.
176  // To get monthly statistics, daily publications should be set too.
177  if(m_monthlyPublication && m_monthlyInfoFile)
178  {
179  recordMonthlyStatistics();

182 
183  }

189  closeTransactionPublication();
190  if(m_signalPublication)
191  closeSignalPublication();
192  std::cout<<"**************************************************"<<std::endl;
193

485 void Simulator::makeTrade( TradingStock &a_stock, double &a_currentPrice, double &a_signalToday )
486 {
487  // Get signal record
488  std::vector<double> t_signals = a_stock.GetSignalRecord();
489 
490  // Check if it is the first trade
491  if(t_signals.size()<2)
492  {
493  // Initial buy/sell only if entry/exit threshold is met respectively.
494  if(a_signalToday>m_entrySignal || a_signalToday<-m_entrySignal)
495  {
496  openPositions(a_stock, a_currentPrice, a_signalToday);
497  a_stock.RecordPNL(0, 0);
498  }
499  // No action
500  else
501  recordNoActivity(a_stock);
502  }
503  // Consecutive trades
504  else
505  {
506  // Get yesterday's signal to check for center-line crossovers
507  double t_signalYesterday = t_signals.at(t_signals.size()-2);
508 
509  // Flag if any positions are closed
510  bool t_closed = false;
511 
512  // Close positions if exit signal is met.
513  if( (a_stock.IsShort() && a_signalToday>(-m_exitSignal) && a_signalToday<0 && a_stock.CheckShortSignal(a_currentPrice))
514  || (a_stock.IsLong() && a_signalToday<m_exitSignal && a_signalToday>0 && a_stock.CheckLongSignal(a_currentPrice)) )
515  {
516  closePositions(a_stock, a_currentPrice, a_signalToday);
517  t_closed = true;
518  }
519 
520  // Check for crossovers of signal for buy/sell indication:
521  // Price will increase, go long
522  if(a_signalToday>0 && t_signalYesterday<0 && a_signalToday<m_entrySignal && hasEnoughCapital(a_stock, a_currentPrice))
523  {
524  openPositions(a_stock, a_currentPrice, a_signalToday);
525 
526  // Record no PNL for the day
527  if(!t_closed)
528  a_stock.RecordPNL(0, 0);
529  }
530  // Price will fall, sell stocks/go short
531  else if(a_signalToday<0 && t_signalYesterday>0 && a_signalToday>(-m_entrySignal) && hasEnoughCapital(a_stock, a_currentPrice))
532  {
533  openPositions(a_stock, a_currentPrice, a_signalToday);
534 
535  // Record no PNL for the day
536  if(!t_closed)
537  a_stock.RecordPNL(0, 0);
538  }
539  // No action
540  else if(!t_closed)
541  recordNoActivity(a_stock);
542 
543  // Close positions that exceed max days limit and record any closed transactions.
544  a_stock.CloseStalePositions(m_maxDaysPerPosition, m_today, a_signalToday, a_currentPrice);
545  if(m_transactionPublication)
546  recordTransaction(a_stock);
547  }
548 }

588 void Simulator::openPositions( TradingStock &a_stock, double &a_currentPrice, double &a_signal )
589 {
590  int t_numOfPositions = Utilities::RoundOff((a_stock.GetAvailableCapital()/a_currentPrice));
591 
592  // Check if opening positions will exceed position limit. (MAX POSITIONS PER STOCK)
593  // Reduce number of positions to be under the limit.
594  int t_positionsHeld = -a_stock.GetShortSharesHeld()+a_stock.GetLongSharesHeld();
595  if( (t_positionsHeld+t_numOfPositions) > m_maxPositionsPerStock)
596  {
597  t_numOfPositions = m_maxPositionsPerStock - t_positionsHeld;
598  if(t_numOfPositions<=0)
599  return;
600  }
601 
602  // No position needs to be open
603  if(t_numOfPositions==0)
604  return;
605 
606  // Open long positions
607  if(a_signal>0)
608  {
609  a_stock.SetLong(true);
610  a_stock.OpenPosition(m_today, a_signal, t_numOfPositions, a_currentPrice);
611  a_stock.BuyShares(t_numOfPositions, a_currentPrice);
612  }
613  // Open short positions
614  else if(a_signal<0)
615  {
616  a_stock.SetShort(true);
617  a_stock.OpenPosition(m_today, a_signal, -t_numOfPositions, a_currentPrice);
618  a_stock.SellShares(t_numOfPositions, a_currentPrice);
619  }
620 }

631 void Simulator::closePositions( TradingStock &a_stock, double &a_currentPrice, double &a_signal )
632 {
633  // Variable passed by reference to get invested capital
634  double t_investedCapital = 0;
635  // Number of positions in stock
636  double t_positions = 0;
637 
638  // Close short positions
639  if(a_signal<0)
640  {
641  // Close short transactions, record pnl and buy borrowed shares
642  a_stock.RecordPNL( -(a_stock.CloseShortPositions(m_today, a_signal, a_currentPrice, t_investedCapital, t_positions)), t_investedCapital);
643  if(m_transactionPublication)
644  recordTransaction(a_stock);
645  a_stock.BuyShares(-t_positions, a_currentPrice);
646  a_stock.UpdateStockPosition();
647  }
648  // Close long positions
649  else
650  {
651  // Close long transactions, record pnl and sell shares
652  a_stock.RecordPNL( (a_stock.CloseLongPositions(m_today, a_signal, a_currentPrice, t_investedCapital, t_positions)), t_investedCapital);
653  if(m_transactionPublication)
654  recordTransaction(a_stock);
655  a_stock.SellShares(t_positions, a_currentPrice);
656  a_stock.UpdateStockPosition();
657  }
658 }

667 double Simulator::calculateSharpeRatio()
668 {
669  std::vector<double> t_dailyReturns(m_dailyCumulativeROR.begin(), m_dailyCumulativeROR.end());
670  double t_averageReturn = Utilities::GetAverage(t_dailyReturns);
671  double t_stddev = Utilities::GetStandardDeviation(t_dailyReturns, t_averageReturn);
672  m_sharpeRecord.push_back( (252/sqrt(252)) * (t_averageReturn / t_stddev) );
673  return m_sharpeRecord.at(m_sharpeRecord.size()-1);
674 }

770 void Simulator::recordMonthlyStatistics()
771 {
772  // Temp value holding variables
773  DateTime* t_date;
774  double t_shortPos;
775  double t_longPos;
776  double t_totalPos;
777  double t_monthlyPNL = 0;
778  double t_cumPNL;
779  double t_ROR = 0;
780  double t_marketValue;
781  double t_netMarketValue;
782  double t_sharpe;
783 
784  // Current month and year
785  int t_currentMonth = 0;
786  int t_currentYear = 0;
787 
788  // File name
789  std::string t_dailyFile = "./publications/DailyReport"+std::to_string(m_simulationCount-1)+".csv";
790 
791  // Validate file open
792  std::ifstream t_ins(t_dailyFile);
793  std::string t_buffer;
794 
795  if(!t_ins)
796  {
797  std::cerr <<"Dorothy Error (recordMonthlyStatistics): Unable to make monthly publication from "<< "["<<t_dailyFile<<"]"<<std::endl;
798  exit(1);
799  }
800 
801  // Skip column headers
802  getline(t_ins, t_buffer);
803 
804  // Parse data
805  while(!t_ins.eof())
806  {
807  // New line
808  t_buffer.clear();
809  getline(t_ins, t_buffer);
810 
811  // Parsing csv data lines
812  std::istringstream t_iss(t_buffer);
813  std::string t_element;
814  int i = 0;
815  while (getline(t_iss, t_element, ','))
816  {
817  // FILL NA value and trim whitespace
818  if(t_element.empty())
819  t_element = std::to_string(-999.0);
820  if(t_element.find(" ") != std::string::npos)
821  Utilities::trimBlanks(t_element);
822 
823  // Record column information:
824  switch(i)
825  {
826  case 0:
827  t_date = new DateTime(stoi(t_element));
828  // Uninitialized current date
829  if(t_currentYear==0 && t_currentMonth==0)
830  {
831  t_currentMonth = t_date->GetMonth();
832  t_currentYear = t_date->GetYear();
833  break;
834  }
835  // Check if same month and year
836  else if(t_date->GetMonth() == t_currentMonth && t_date->GetYear() == t_currentYear)
837  break;
838  // Next month reached.
839  else
840  {
841  // Record data gathered which is the statistics on the last day of the month
842  m_monthlyInfoFile << std::to_string(t_currentMonth)+"/"+ std::to_string(t_currentYear) << "," \
843  << t_shortPos <<"," \
844  << t_longPos <<"," \
845  << t_totalPos <<"," \
846  << t_monthlyPNL <<"," \
847  << t_cumPNL <<"," \
848  << t_ROR <<"," \
849  << t_marketValue <<"," \
850  << t_netMarketValue <<","\
851  << t_sharpe<<"\n";
852 
853  // Update to next month and reset values
854  t_currentMonth = t_date->GetMonth();
855  t_currentYear = t_date->GetYear();
856  t_monthlyPNL = 0;
857  t_ROR = 0;
858  break;
859  }
860  case 1:
861  break;
862  case 2:
863  t_shortPos = std::stod(t_element);
864  break;
865 
866  case 3:
867  t_longPos = std::stod(t_element);
868  break;
869 
870  case 4:
871  t_totalPos = std::stod(t_element);
872  break;
873 
874  case 5:
875  t_monthlyPNL += std::stod(t_element);
876  break;
877 
878  case 6:
879  t_cumPNL = std::stod(t_element);
880  break;
881 
882  case 7:
883  t_ROR += std::stod(t_element);
884  break;
885 
886  case 8:
887  t_marketValue = std::stod(t_element);
888  break;
889 
890  case 9:
891  t_netMarketValue = std::stod(t_element);
892  break;
893 
894  case 10:
895  t_sharpe = std::stod(t_element);
896  break;
897 
898  }
899  i++;
900  }
901  }
902 
903  // Record the last month:
904  m_monthlyInfoFile << std::to_string(t_currentMonth)+"/"+ std::to_string(t_currentYear) << "," \
905  << t_shortPos <<"," \
906  << t_longPos <<"," \
907  << t_totalPos <<"," \
908  << t_monthlyPNL <<"," \
909  << t_cumPNL <<"," \
910  << t_ROR <<"," \
911  << t_marketValue <<"," \
912  << t_netMarketValue <<","\
913  << t_sharpe<<"\n";
914 }
